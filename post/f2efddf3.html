  • 关于创建表头、插入数据、写入表格
private static void writeObjectToExcel(List<Object[]> list, String[] excelTitle, String fileName) {

        //在内存中创建Excel文件
        Workbook workbook;
        if (fileName.endsWith("xls")) {
            workbook = new HSSFWorkbook();
        } else if (fileName.endsWith("xlsx")) {
            workbook = new XSSFWorkbook();
        } else {
            throw new IllegalArgumentException("fileName not legal");
        }
        Sheet sheet = workbook.createSheet();
        //标题行
        Row titleRow = sheet.createRow(0);
        for (int i = 0; i < excelTitle.length; i++) {
            titleRow.createCell(i).setCellValue(excelTitle[i]);
        }
        //创建数据行并写入值
        for (Object[] dataArr : list) {
            int lastRowNum = sheet.getLastRowNum();
            Row dataRow = sheet.createRow(lastRowNum + 1);
            System.out.println("插入的数据："+JSON.toJSONString(dataArr));
            for (int i = 0; i < dataArr.length; i++) {
                Cell cell = dataRow.createCell(i);
                sheet.autoSizeColumn(i,true);
                Object cellValue = dataArr[i];
                if(cellValue != null) {
                    setCellValue(cellValue, cell);
                }
            }
        }
        //创建输出流对象
        FileOutputStream outputStream = null;
        try {
            outputStream = new FileOutputStream(new File(fileName));
        } catch (FileNotFoundException e) {
            log.error("file not found", e);
        }
        try {
            workbook.write(outputStream);
        } catch (IOException e) {
            log.error("write to file failed", e);
        } finally {
            if (outputStream != null) {
                try {
                    outputStream.close();
                } catch (IOException ignore) {
                }
            }
        }
    }

二、EasyExcel

官方文档参考地址：https://www.yuque.com/easyexcel/doc/easyexcel

官方介绍：EasyExcel是一个基于Java的简单、省内存的读写Excel的开源项目。在尽可能节约内存的情况下支持读写百M的Excel。

个人小案例地址:https://gitee.com/sky508/easyexcel-demo.git

  • 先导入依赖

            <!--阿里巴巴EasyExcel-->
        <dependency>
            <groupId>com.alibaba</groupId>
            <artifactId>easyexcel</artifactId>
            <version>2.1.6</version>
        </dependency>

        <!--xml需要的-->

        <dependency>

        </dependency>

  • 设置表头，封装需要写入表格的数据的数据格式，然后构建一些表格基础信息，就行了

        public static void main(String[] args){

            //有9682个xml文件，需要提取数据并把相关数据插入表格
            int num =9682;

            for (int i =1;i<=num;i++){

                XmlToExcelUtil3 xmlToExcelUtil3 = new XmlToExcelUtil3();

                //需要插入表格的数据
                List<List<Object>> devs = xmlToExcelUtil3.toExcel(i,"D:/软件下载/xml/down/exportDeviceTemplate_typeid="+i+".xml");

                if(i == num){

                    String pathName = "E:/oid.xlsx";
                    String sheetName = "oid表";

                    //EasyExcel写法
                    EasyExcel.write(pathName).head(head()).sheet(sheetName).registerWriteHandler(new LongestMatchColumnWidthStyleStrategy()).doWrite(devs);
                }
            }
        }
//设置表头
    private static List<List<String>> head(){

        List<String> head3 = new ArrayList<>();
        head3.add("oid");
        head3.add("CategoryName");

        List<String> head4 = new ArrayList<>();
        head4.add("oid");

        objects.add(head0);
        objects.add(head1);
        objects.add(head2);

    }
